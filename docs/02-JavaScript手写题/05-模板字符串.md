# 实现模板字符串

```js
let str = "My name is ${name}, I am from ${city}",
  info = {
    name: "AaDerBrane",
    city: "GungZhou",
  };
// My name is AaDerBrane, I am from GuangZhou
function printf(str, info) {
  str.replace(/\${([^{}]+)}/g, (matchStr, word, index) => {
    for (let item in info) {
      str = str.replace("${" + word + "}", info[item]);
    }
  });
  return str;
}

console.log("printf(str, info) ", printf(str, info));
```
